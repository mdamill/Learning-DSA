 ## KMP Algorithm :-

 ### Problem : Longest Prefix Suffix

```
Examples :

Input: str = "abab"
Output: 2
Explanation: "ab" is the longest proper prefix and suffix. 

Input: str = "aaaa"
Output: 3
Explanation: "aaa" is the longest proper prefix and suffix. 
```


 ```cpp
int lps(string str) {
       
       vector<int> lps(str.size(),0);
       int pre = 0, suf = 1;
       
       while(suf < str.size()){
           
           if(str[pre]==str[suf])
           {
               lps[suf] = pre+1;
               suf++;
               pre++;
           }
           else
           {
               // Match
               if(pre == 0){
                   lps[suf] = 0;
                   suf++;
               }
               // !match
               else{
                   pre = lps[pre-1];
               }
           }
           
       }
       return lps[lps.size()-1];
    }
 ```
